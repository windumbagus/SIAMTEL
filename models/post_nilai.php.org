	function A($disiplin,$sopan,$personal,$tim,$adaptasi,$tj,$jujur){
		$a =($disiplin+$sopan+$personal+$tim+$adaptasi+$tj+$jujur)/7;
		
				if ($a > 80) {
				$n_akhir = 'A';
				}elseif ($a >= 70 && $a <80 ) {
				$n_akhir = 'B';
				}elseif ($a >=60 && $a <70 ) {
				$n_akhir = 'C';
				}elseif ( $a >=50&& $a <60 ) {
				$n_akhir = 'D';
				}elseif ( $a <50 ) {
				$n_akhir = 'E';
				}
				return $n_akhir;
	}
